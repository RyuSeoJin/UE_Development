02 — 설계 결정

왜 이렇게 만드나

카테고리 2축 모델 — 대분류는 길드에, 소분류는 계정에

카테고리가 두 축으로 나뉩니다. 대분류(guilds.Category)는 길드의 속성으로 길드당 1개(길드 간 중복 허용), 소분류(Archive_Category)는 모든 길드가 공유하는 계정 풀로 기록마다 태그됩니다.

예시대분류 (guilds.Category)소분류 태그 가능
A 길드캐릭터'애니메이션' 등 — 공유 풀에서 어느 길드든 사용
B 길드배경
C 길드캐릭터 (A와 중복 OK)

필터 "캐릭터"(대분류) → A·C 길드의 기록. 필터 "애니메이션"(소분류) → A·B·C 전체의 해당 기록. 이후 필터 UI가 어느 축으로 검색할지 상세 구분·조합할 예정이므로, 4단계의 조회 SQL을 축별 조각으로 조립할 수 있게 설계합니다.

GuildId 발급 — 왜 AUTOINCREMENT인가 (상세)

SQLite에서 INTEGER PRIMARY KEY와 INTEGER PRIMARY KEY AUTOINCREMENT는 비슷해 보이지만 번호 재사용 규칙이 다릅니다.

구분INTEGER PRIMARY KEY (일반)+ AUTOINCREMENT
번호 발급현재 존재하는 최대 rowid + 1역대 최대값 + 1 (sqlite_sequence 테이블에 기록)
행 삭제 후⚠️ 마지막 행을 지우면 그 번호가 재사용될 수 있음✅ 지워도 영구히 재사용 안 됨(단조 증가)
비용—sqlite_sequence 1행 관리 — 이 규모에선 무시 가능

왜 재사용 금지가 중요한가 — archives의 모든 기록이 GuildId로 태그되고 기록 ID 접미사에도 길드번호가 박힙니다(07 문서). 길드를 완전 삭제(Purge)한 뒤 그 번호가 새 길드에 재발급되면, 과거 기록이 남의 길드 것으로 오인됩니다. AUTOINCREMENT는 이 금지를 코드가 아니라 DB 엔진이 보장하게 합니다.

💡

부수 효과 두 가지. ① 기존 설계의 player.NextGuildId 카운터는 미사용이 됩니다 — GuildSubsystem이 Economy에 번호를 빌리러 갈 필요가 없어져 결합이 사라집니다(05 개정 반영). ② DB Browser에 sqlite_sequence라는 테이블이 새로 보이는데, 이건 AUTOINCREMENT의 장부입니다 — 지우면 번호가 되감겨 재사용 금지가 깨지니 절대 삭제 금지.

// 길드 하나 — guilds 1행의 메모리 캐시. 캐릭터·Gold는 여기 없음(계정 공유)
USTRUCT(BlueprintType)
struct FGuildData
{
    GENERATED_BODY()
    UPROPERTY(BlueprintReadOnly) int32 GuildId = 0;        // AUTOINCREMENT 발급(재사용 없음)
    UPROPERTY(BlueprintReadOnly) FString GuildName;
    UPROPERTY(BlueprintReadOnly) FString Category;         // 대분류 1개(길드 간 중복 허용) — 구 Categories 배열 대체
    UPROPERTY(BlueprintReadOnly) int32 FocusSeconds = 0;   // 레벨의 원천(기록 삭제와 무관하게 유지)
    UPROPERTY(BlueprintReadOnly) FDateTime CreatedAt;       // UTC

CreateGuild + 목록 로드
INSERT → 엔진이 준 새 번호 받기(GetLastInsertRowId)
int32 UGuildSubsystem::CreateGuild(const FString& Name, const FString& Category)
{
    if (!Db || !Db->IsValid() || Name.IsEmpty()) return -1;

    FSQLitePreparedStatement Stmt;
    // GuildId는 넣지 않는다 — AUTOINCREMENT가 발급(넣으면 엔진 보장을 스스로 깨는 셈)
    Stmt.Create(*Db, TEXT(
        "INSERT INTO guilds (GuildName, Category, CreatedAt) VALUES (?1, ?2, ?3);"));
    Stmt.SetBindingValueByIndex(1, Name);
    Stmt.SetBindingValueByIndex(2, Category);
    Stmt.SetBindingValueByIndex(3, FDateTime::UtcNow().ToIso8601()); // UTC 고정(02 결정)
    const bool bOk = Stmt.Execute();
    Stmt.Destroy();
    if (!bOk) return -1;

    // 방금 INSERT가 받은 번호 = 이 연결의 마지막 rowid. 이걸 캐시·UI가 쓴다
    const int32 NewId = (int32)Db->GetLastInsertRowId();
    LoadGuilds();                     // 캐시 재적재(단순·확실 — 길드 수가 적어 비용 무시 가능)
    return NewId;
}

void UGuildSubsystem::LoadGuilds()
{
    Guilds.Reset();
    FSQLitePreparedStatement Stmt;
    // 살아있는 길드만(DeletedAt IS NULL), UI 정렬 순서 → 생성 순 보조 정렬
    Stmt.Create(*Db, TEXT(
        "SELECT GuildId, GuildName, Category, FocusSeconds, CreatedAt, SortOrder, Icon, Secured "
        "FROM guilds WHERE DeletedAt IS NULL ORDER BY SortOrder, GuildId;"));
    while (Stmt.Step() == ESQLitePreparedStatementStepResult::Row)
    {
        FGuildData G; FString IsoTime; TArray<uint8> Blob;
        Stmt.GetColumnValueByIndex(0, G.GuildId);
        Stmt.GetColumnValueByIndex(1, G.GuildName);
        Stmt.GetColumnValueByIndex(2, G.Category);
        Stmt.GetColumnValueByIndex(3, G.FocusSeconds);
        Stmt.GetColumnValueByIndex(4, IsoTime);
        FDateTime::ParseIso8601(*IsoTime, G.CreatedAt);   // TEXT → FDateTime
        Stmt.GetColumnValueByIndex(5, G.SortOrder);
        Stmt.GetColumnValueByIndex(6, G.Icon);
        Stmt.GetColumnValueByIndex(7, Blob);              // Secured — 검증은 6단계에서 이 자리에 추가
        Guilds.Add(G);
    }
    Stmt.Destroy();
}
4
2축 카테고리 필터 SQL
대분류=길드를 거쳐, 소분류=기록을 직접 — 조합도 가능

두 축은 거치는 테이블이 다릅니다 — 대분류는 guilds를 JOIN해서 거르고, 소분류는 archives.Category를 직접 거릅니다. 이후 상세 필터 UI가 "어느 축으로 검색할지"를 고르거나 두 축을 조합할 예정이므로, WHERE 조각을 축별로 조립하는 구조를 권장합니다.

-- 공통 뼈대: 기록 + 소속 길드. 살아있는 것만(길드가 휴지통이면 그 기록도 자동으로 숨음)
SELECT a.* FROM archives a
JOIN guilds g ON g.GuildId = a.GuildId
WHERE g.DeletedAt IS NULL AND a.DeletedAt IS NULL
-- ▼ 아래 조각을 필터 UI 선택에 따라 덧붙인다 ▼

-- 축 1) 대분류(Guild_Category) 필터: '캐릭터' → A·C 길드의 기록
  AND g.Category = '캐릭터'

-- 축 2) 소분류(Archive_Category) 필터: '애니메이션' → 전 길드에서 태그된 기록
  AND a.Category = '애니메이션'

-- 조합) 상세 필터: 두 축을 함께 — '캐릭터' 길드들의 '애니메이션' 기록만
  AND g.Category = '캐릭터' AND a.Category = '애니메이션'

ORDER BY a.CreatedAt DESC;   -- 정렬은 항상 CreatedAt 컬럼(문자열 ID 아님 — 07 규약)
💡

코드에서는 — 조회 함수를 GetRecords(옵션: 대분류, 소분류) 하나로 두고, 넘어온 옵션에 따라 위 WHERE 조각을 바인딩으로 덧붙이면 상세 필터 UI가 어떤 조합을 요구해도 SQL 하나로 대응됩니다.

소분류 풀 관리 — 첫 실전 트랜잭션
이름 변경은 풀+기록 두 테이블을 한 덩어리로

추가는 한 테이블이라 단순하지만, 이름 변경은 두 테이블(풀의 정의 + 기록의 태그)을 함께 고쳐야 합니다 — 03·05 문서에서 설계로만 보던 BEGIN…COMMIT이 처음 실전 투입되는 지점입니다.

void UGuildSubsystem::AddArchiveCategory(const FString& Name, FLinearColor Color, const FString& Icon)
{
    if (!Db || !Db->IsValid() || Name.IsEmpty()) return;
    FSQLitePreparedStatement Stmt;
    // OR IGNORE: 같은 이름이 이미 있으면 조용히 무시(Name이 PK라 중복은 어차피 거부됨)
    Stmt.Create(*Db, TEXT(
        "INSERT OR IGNORE INTO archive_categories (Name, Color, Icon, CreatedAt) VALUES (?1, ?2, ?3, ?4);"));
    Stmt.SetBindingValueByIndex(1, Name);
    // SQLite엔 색 타입이 없다 — FLinearColor → "RRGGBBAA" 헥스 문자열로 변환해 TEXT 컬럼에
    Stmt.SetBindingValueByIndex(2, Color.ToFColor(true).ToHex());
    Stmt.SetBindingValueByIndex(3, Icon);
    Stmt.SetBindingValueByIndex(4, FDateTime::UtcNow().ToIso8601());
    Stmt.Execute();
    Stmt.Destroy();
}

void UGuildSubsystem::RenameArchiveCategory(const FString& OldName, const FString& NewName)
{
    if (!Db || !Db->IsValid() || OldName == NewName || NewName.IsEmpty()) return;

    // 풀(이름 정의)과 기록(태그)이 어긋나면 필터가 깨진다 — 두 UPDATE를 한 덩어리로 묶는다.
    // 중간에 크래시해도 COMMIT 전이면 둘 다 원상 복귀: "풀만 새 이름, 기록은 옛 이름" 상태가 불가능
    Db->Execute(TEXT("BEGIN TRANSACTION;"));
    {
        FSQLitePreparedStatement S1;
        S1.Create(*Db, TEXT("UPDATE archive_categories SET Name = ?1 WHERE Name = ?2;"));
        S1.SetBindingValueByIndex(1, NewName); S1.SetBindingValueByIndex(2, OldName);
        S1.Execute(); S1.Destroy();

        FSQLitePreparedStatement S2;
        S2.Create(*Db, TEXT("UPDATE archives SET Category = ?1 WHERE Category = ?2;"));
        S2.SetBindingValueByIndex(1, NewName); S2.SetBindingValueByIndex(2, OldName);
        S2.Execute(); S2.Destroy();
    }
    Db->Execute(TEXT("COMMIT;"));
}

TArray<FArchiveCategory> UGuildSubsystem::GetArchiveCategories() const
{
    TArray<FArchiveCategory> Out;
    if (!Db || !Db->IsValid()) return Out;
    FSQLitePreparedStatement Stmt;
    // 풀 테이블이 정본 — 기록이 없는 "빈 카테고리"도 목록에 나온다(풀 채택의 이유)
    Stmt.Create(*Db, TEXT("SELECT Name, Color, Icon FROM archive_categories ORDER BY SortOrder, Name;"));
    while (Stmt.Step() == ESQLitePreparedStatementStepResult::Row)
    {
        FArchiveCategory C; FString Hex;
        Stmt.GetColumnValueByIndex(0, C.Name);
        Stmt.GetColumnValueByIndex(1, Hex);
        Stmt.GetColumnValueByIndex(2, C.Icon);
        // "RRGGBBAA" 헥스 TEXT → FLinearColor 복원(저장 때의 역변환)
        C.Color = FLinearColor::FromSRGBColor(FColor::FromHex(Hex));
        Out.Add(C);
    }
    Stmt.Destroy();
    return Out;
}

시간 적립 + Secured — 블롭 스왑까지 막기
페이로드에 GuildId를 넣어 "남의 블롭 붙여넣기"를 거부

11의 RTSecured에 길드용 봉인/개봉을 추가합니다. 핵심 차이: player는 행이 하나지만 길드는 여러 행이라, 시간만 봉인하면 "시간 많은 길드의 블롭을 다른 길드 행에 복사"하는 치트가 통합니다. 페이로드에 GuildId를 함께 봉인해 행과 블롭을 묶습니다.

// 길드 시간 봉인 — GuildId를 페이로드에 포함해 행↔블롭을 묶는다(스왑 치트 차단)
TArray<uint8> SealGuildTime(int32 GuildId, int64 FocusSeconds);

// 개봉 — MAGIC·CRC에 더해 "이 행의 GuildId가 맞는지"까지 검사. 하나라도 어긋나면 false
bool OpenGuildTime(const TArray<uint8>& Blob, int32 ExpectedGuildId, int64& OutFocusSeconds);
// 길드용 서명 — 계정용(RTA1)과 다른 값이라 서로의 블롭을 바꿔치기해도 개봉 실패
static constexpr uint32 GUILD_MAGIC = 0x52544731;   // 'RTG1'

TArray<uint8> SealGuildTime(int32 GuildId, int64 FocusSeconds)
{
    // 페이로드 = MAGIC | GuildId | FocusSeconds | CRC — 이하 AES 절차는 SealAccount(11)와 동일
    FBufferArchive Ar;
    uint32 Magic = GUILD_MAGIC;
    Ar << Magic << GuildId << FocusSeconds;           // 4+4+8 = 16바이트
    uint32 Crc = FCrc::MemCrc32(Ar.GetData(), Ar.Num());
    Ar << Crc;
    const int32 Pad = (16 - (Ar.Num() % 16)) % 16;
    for (int32 i = 0; i < Pad; ++i) { uint8 Z = 0; Ar << Z; }
    FAES::FAESKey K; MakeKey(K);                       // 11과 같은 키 재사용
    FAES::EncryptData(Ar.GetData(), Ar.Num(), K);
    TArray<uint8> Out; Out.Append(Ar.GetData(), Ar.Num());
    return Out;
}

bool OpenGuildTime(const TArray<uint8>& Blob, int32 ExpectedGuildId, int64& OutFocusSeconds)
{
    if (Blob.Num() == 0 || Blob.Num() % 16 != 0) return false;
    TArray<uint8> Buf = Blob;
    FAES::FAESKey K; MakeKey(K);
    FAES::DecryptData(Buf.GetData(), Buf.Num(), K);

    FMemoryReader Ar(Buf);
    uint32 Magic = 0; Ar << Magic;
    if (Magic != GUILD_MAGIC) return false;          // 키 틀림/변조/계정 블롭 바꿔치기
    int32 SealedGuildId = 0; int64 Focus = 0;
    Ar << SealedGuildId << Focus;
    uint32 StoredCrc = 0; Ar << StoredCrc;
    if (FCrc::MemCrc32(Buf.GetData(), 16) != StoredCrc) return false;
    if (SealedGuildId != ExpectedGuildId) return false; // ★ 스왑 방지 — 다른 행의 블롭이면 여기서 거부

    OutFocusSeconds = Focus;
    return true;
}
#include "RTSecured.h"

void UGuildSubsystem::AddFocusSeconds(int32 GuildId, int32 Seconds)
{
    if (Seconds <= 0 || !Db || !Db->IsValid()) return;
    FGuildData* G = Guilds.FindByPredicate([=](const FGuildData& X){ return X.GuildId == GuildId; });
    if (!G) return;                       // 휴지통 길드엔 적립 불가(캐시=살아있는 것만)
    G->FocusSeconds += Seconds;

    // 평문+봉인을 같은 UPDATE로 — 11의 SavePlayer와 동일한 원칙
    const TArray<uint8> Sealed = SealGuildTime(GuildId, G->FocusSeconds);
    FSQLitePreparedStatement Stmt;
    Stmt.Create(*Db, TEXT("UPDATE guilds SET FocusSeconds = ?1, Secured = ?2 WHERE GuildId = ?3;"));
    Stmt.SetBindingValueByIndex(1, G->FocusSeconds);
    Stmt.SetBindingValueByIndex(2, Sealed);
    Stmt.SetBindingValueByIndex(3, GuildId);
    Stmt.Execute();
    Stmt.Destroy();
}

소프트 삭제 / 복원
행은 남기고 DeletedAt만 — 기록은 JOIN이 자동으로 숨긴다
void UGuildSubsystem::SoftDeleteGuild(int32 GuildId)
{
    if (!Db || !Db->IsValid()) return;
    FSQLitePreparedStatement Stmt;
    // 행 삭제가 아니라 시각 기록 — 07 규약. 이 길드의 기록 행은 건드리지 않는다:
    // 4단계 조회 SQL의 "g.DeletedAt IS NULL" 조건이 그 기록들을 자동으로 함께 숨긴다(복원하면 함께 복귀)
    Stmt.Create(*Db, TEXT("UPDATE guilds SET DeletedAt = ?1 WHERE GuildId = ?2;"));
    Stmt.SetBindingValueByIndex(1, FDateTime::UtcNow().ToIso8601());
    Stmt.SetBindingValueByIndex(2, GuildId);
    Stmt.Execute();
    Stmt.Destroy();
    LoadGuilds();   // 캐시에서 사라짐(살아있는 것만 담으므로)
}

void UGuildSubsystem::RestoreGuild(int32 GuildId)
{
    if (!Db || !Db->IsValid()) return;
    FSQLitePreparedStatement Stmt;
    Stmt.Create(*Db, TEXT("UPDATE guilds SET DeletedAt = NULL WHERE GuildId = ?1;")); // 되살림 = NULL
    Stmt.SetBindingValueByIndex(1, GuildId);
    Stmt.Execute();
    Stmt.Destroy();
    LoadGuilds();
}
