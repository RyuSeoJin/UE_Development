저장 전략 — 풀-DB, 파일 하나

영속 데이터(껐다 켜도 남아야 하는 것)는 전부 SQLite 파일 하나([프로젝트]/Saved/Archive.db)에 삽니다. SaveGame(.sav)은 쓰지 않습니다. 파일 안에서 상태(재화·로스터·길드·건물)는 행을 UPDATE하고, 기록(아카이브)은 행을 INSERT로 쌓습니다. 파일이 하나라서 ① 저장소 정합성 걱정이 없고 ② 여러 테이블을 함께 고치는 작업을 트랜잭션(BEGIN…COMMIT) 하나로 묶을 수 있으며 ③ DB Browser for SQLite(무료 도구)로 모든 데이터를 눈으로 확인할 수 있습니다.

① RTTypes.h — 공용 그릇(타입 계층)

DB 행을 메모리로 옮길 때 담는 틀입니다. 여러 시스템·UI가 공유하므로 중립 파일에 모아둡니다.

// RTTypes.h — DB 행을 담는 공용 그릇(struct·enum)의 단일 보관함. 로직 없음
#pragma once
#include "CoreMinimal.h"
#include "RTTypes.generated.h"   // UENUM·USTRUCT를 쓰므로 필요 — 항상 맨 마지막 include

// 기록이 어디서 왔나 — archives.Source 컬럼(INTEGER)에 대응.
// 순서를 바꾸면 기존 DB의 숫자와 어긋나므로 새 항목은 반드시 뒤에만 추가할 것
UENUM(BlueprintType)
enum class ERecordSource : uint8
{
    Timer,       // 집중 타이머
    Stopwatch,   // 스톱워치
    Quest        // 퀘스트 완료
};

// 아카이브 기록 하나 — archives 1행. GetArchives는 살아있는 기록만 담아 반환
USTRUCT(BlueprintType)
struct FArchiveRecord
{
    GENERATED_BODY()
    UPROPERTY(BlueprintReadOnly) FString Id;                    // "YYYYMMDDhhmmssSSS-{GUILDID}" — 발급 후 불변
    UPROPERTY(BlueprintReadOnly) int32 GuildId = 0;            // 길드 스코프 태그(필터 키)
    UPROPERTY(BlueprintReadOnly) FString Category;              // 소분류(계정 공유 풀의 이름)
    UPROPERTY(BlueprintReadOnly) ERecordSource Source = ERecordSource::Timer;
    UPROPERTY(BlueprintReadOnly) FDateTime CreatedAt;           // 정렬 정본(문자열 Id로 정렬 금지)
    UPROPERTY(BlueprintReadOnly) int32 DurationSeconds = 0;
    UPROPERTY(BlueprintReadOnly) FString Title;
};

// 소분류 카테고리 하나 — archive_categories 1행. 이름·색·아이콘이 UI에 항상 쌍으로 다닌다
USTRUCT(BlueprintType)
struct FArchiveCategory
{
    GENERATED_BODY()
    UPROPERTY(BlueprintReadOnly) FString Name;         // 플레이어가 정한 이름(PK)
    UPROPERTY(BlueprintReadOnly) FLinearColor Color = FLinearColor::White; // DB엔 "RRGGBBAA" 헥스 TEXT로 저장
    UPROPERTY(BlueprintReadOnly) FString Icon;         // 플레이어가 고른 아이콘 식별자
};

// 길드 하나 — guilds 1행. 캐릭터·Gold는 여기 없음(계정 공유). 레벨은 파생(저장 안 함)
USTRUCT(BlueprintType)
struct FGuildData
{
    GENERATED_BODY()
    UPROPERTY(BlueprintReadOnly) int32 GuildId = 0;        // AUTOINCREMENT 발급(재사용 없음)
    UPROPERTY(BlueprintReadOnly) FString GuildName;
    UPROPERTY(BlueprintReadOnly) FString Category;         // 대분류 1개(길드 간 중복 허용) — 2축 모델
    UPROPERTY(BlueprintReadOnly) int32 FocusSeconds = 0;   // 레벨의 원천(기록 삭제와 무관하게 유지)
    UPROPERTY(BlueprintReadOnly) FDateTime CreatedAt;       // UTC
    UPROPERTY(BlueprintReadOnly) int32 SortOrder = 0;      // UI 정렬

bool UDatabaseSubsystem::BeginTransaction()
{
    // DB가 안 열렸거나 이미 트랜잭션 중이면 거절 — 중첩 BEGIN은 SQLite 에러
    if (!Database.IsValid() || bInTransaction) return false;
    bInTransaction = Database.Execute(TEXT("BEGIN TRANSACTION;"));
    return bInTransaction;
}

bool UDatabaseSubsystem::CommitTransaction()
{
    if (!Database.IsValid() || !bInTransaction) return false;   // 시작한 적 없으면 확정할 것도 없다
    bInTransaction = false;
    return Database.Execute(TEXT("COMMIT;"));   // 묶인 변경 전부를 이 순간 한 번에 확정
}

bool UDatabaseSubsystem::RollbackTransaction()
{
    if (!Database.IsValid() || !bInTransaction) return false;
    bInTransaction = false;
    return Database.Execute(TEXT("ROLLBACK;"));   // BEGIN 이후의 모든 변경을 되돌린다
}

③ RTSecured — 치트 방어 봉인/개봉(인프라 계층)

치트 유인이 큰 값(계정 재화·시간, 길드 시간)을 암호화 사본(BLOB)으로 함께 저장해, DB Browser로 평문을 고치면 로드 때 자동으로 되돌립니다. GKey 32바이트는 반드시 본인이 무작위 값으로 채우세요.

// RTSecured.h — 봉인/개봉 자유 함수(클래스 아님). FAES·FCrc는 Core 모듈이라 Build.cs 추가 불필요
#pragma once
#include "CoreMinimal.h"

// player 행의 치트 유인 값 3개를 하나로 묶은 그릇 — 이 값들만 Secured로 보호한다
struct FSecuredAccount
{
    int64 Gold = 0;
    int32 Shards = 0;
    int64 AccountFocusSeconds = 0;
};

// [계정] 봉인: 값 → (MAGIC+값+CRC) 평문 → AES 암호화 → BLOB. player.Secured에 저장
TArray<uint8> SealAccount(const FSecuredAccount& In);
// [계정] 개봉: BLOB → AES 복호화 → MAGIC·CRC 검사. 성공=true(Out 채움), 실패=false(변조/손상)
bool OpenAccount(const TArray<uint8>& Blob, FSecuredAccount& Out);

// [길드] 봉인 — GuildId를 페이로드에 포함해 행↔블롭을 묶는다(다른 길드 블롭 붙여넣기 = 스왑 치트 차단)
TArray<uint8> SealGuildTime(int32 GuildId, int64 FocusSeconds);
// [길드] 개봉 — MAGIC·CRC에 더해 "이 행의 GuildId가 맞는지"까지 검사. 하나라도 어긋나면 false
bool OpenGuildTime(const TArray<uint8>& Blob, int32 ExpectedGuildId, int64& OutFocusSeconds);
#include "RTSecured.h"
#include "Misc/AES.h"        // FAES::EncryptData/DecryptData (AES-256)
#include "Misc/Crc.h"        // FCrc::MemCrc32 — 무결성 검사
#include "Serialization/BufferArchive.h"
#include "Serialization/MemoryReader.h"

// 빌드에 심는 32바이트(256비트) 대칭키 — 반드시 무작위 값으로 직접 채울 것.
// 키가 클라에 있으므로 "완전 방어"는 아니다(서버 없는 싱글의 적정선)
static const uint8 GKey[32] = { /* ★ 32개 무작위 바이트를 직접 채우세요 ★ */ };
static constexpr uint32 SEAL_MAGIC  = 0x52544131;   // 'RTA1' — 계정 블롭 서명
static constexpr uint32 GUILD_MAGIC = 0x52544731;   // 'RTG1' — 길드 블롭 서명(계정과 다른 값이라 서로 바꿔치기 불가)

static void MakeKey(FAES::FAESKey& K) { FMemory::Memcpy(K.Key, GKey, 32); }

TArray<uint8> SealAccount(const FSecuredAccount& In)
{
    // 1) 평문 페이로드 = MAGIC | Gold | Shards | Seconds | CRC32(앞 24바이트)
    FBufferArchive Ar;
    uint32 Magic = SEAL_MAGIC;
    int64 G = In.Gold; int32 S = In.Shards; int64 F = In.AccountFocusSeconds;
    Ar << Magic << G << S << F;                       // 4+8+4+8 = 24바이트
    uint32 Crc = FCrc::MemCrc32(Ar.GetData(), Ar.Num());
    Ar << Crc;                                        // 무결성 태그(변조 감지)를 뒤에 붙임

    // 2) AES는 16바이트 블록 단위 — 0으로 채워 배수를 맞춘 뒤 제자리 암호화
    const int32 Pad = (16 - (Ar.Num() % 16)) % 16;
    for (int32 i = 0; i < Pad; ++i) { uint8 Z = 0; Ar << Z; }
    FAES::FAESKey K; MakeKey(K);
    FAES::EncryptData(Ar.GetData(), Ar.Num(), K);

    TArray<uint8> Out; Out.Append(Ar.GetData(), Ar.Num());
    return Out;                                       // → player.Secured BLOB
}

bool OpenAccount(const TArray<uint8>& Blob, FSecuredAccount& Out)
{
    // 블롭이 없거나 블록 배수가 아니면 손상 — 개봉 실패
    if (Blob.Num() == 0 || Blob.Num() % 16 != 0) return false;

    TArray<uint8> Buf = Blob;                        // 원본 보존을 위해 복사본을 복호화
    FAES::FAESKey K; MakeKey(K);
    FAES::DecryptData(Buf.GetData(), Buf.Num(), K); // 키가 틀리면 쓰레기 바이트가 나온다

    FMemoryReader Ar(Buf);
    uint32 Magic = 0; Ar << Magic;
    if (Magic != SEAL_MAGIC) return false;           // 서명 불일치 = 키 틀림/블롭 변조

    int64 G = 0; int32 S = 0; int64 F = 0;
    Ar << G << S << F;
    uint32 StoredCrc = 0; Ar << StoredCrc;
    const int32 PayloadBytes = 24;                    // MAGIC(4)+G(8)+S(4)+F(8) — 봉인 때와 동일 범위
    if (FCrc::MemCrc32(Buf.GetData(), PayloadBytes) != StoredCrc) return false; // 무결성 실패

    Out.Gold = G; Out.Shards = S; Out.AccountFocusSeconds = F;
    return true;                                     // 진짜 값 복원 성공
}

TArray<uint8> SealGuildTime(int32 GuildId, int64 FocusSeconds)
{
    // 페이로드 = MAGIC | GuildId | FocusSeconds | CRC — 이하 AES 절차는 SealAccount와 동일
    FBufferArchive Ar;
    uint32 Magic = GUILD_MAGIC;
    Ar << Magic << GuildId << FocusSeconds;           // 4+4+8 = 16바이트
    uint32 Crc = FCrc::MemCrc32(Ar.GetData(), Ar.Num());
    Ar << Crc;
    const int32 Pad = (16 - (Ar.Num() % 16)) % 16;
    for (int32 i = 0; i < Pad; ++i) { uint8 Z = 0; Ar << Z; }
    FAES::FAESKey K; MakeKey(K);                       // 계정과 같은 키 재사용
    FAES::EncryptData(Ar.GetData(), Ar.Num(), K);
    TArray<uint8> Out; Out.Append(Ar.GetData(), Ar.Num());
    return Out;
}

bool OpenGuildTime(const TArray<uint8>& Blob, int32 ExpectedGuildId, int64& OutFocusSeconds)
{
    if (Blob.Num() == 0 || Blob.Num() % 16 != 0) return false;
    TArray<uint8> Buf = Blob;
    FAES::FAESKey K; MakeKey(K);
    FAES::DecryptData(Buf.GetData(), Buf.Num(), K);

    FMemoryReader Ar(Buf);
    uint32 Magic = 0; Ar << Magic;
    if (Magic != GUILD_MAGIC) return false;          // 키 틀림/변조/계정 블롭 바꿔치기
    int32 SealedGuildId = 0; int64 Focus = 0;
    Ar << SealedGuildId << Focus;
    uint32 StoredCrc = 0; Ar << StoredCrc;
    if (FCrc::MemCrc32(Buf.GetData(), 16) != StoredCrc) return false;
    if (SealedGuildId != ExpectedGuildId) return false; // ★ 스왑 방지 — 다른 행의 블롭이면 여기서 거부

    OutFocusSeconds = Focus;
    return true;
}

    FString MakeArchiveId(int32 GuildId) const;   // 기록 ID 발급 규칙
    FSQLiteDatabase* Db = nullptr;                // 문지기에게 빌린 핸들(소유 아님 — Close 금지)
};

#include "DatabaseSubsystem.h"          // 문지기 — GetDb()
#include "SQLiteDatabase.h"             // FSQLiteDatabase 실체(전방 선언 해소)

void UArchiveSubsystem::Initialize(FSubsystemCollectionBase& Collection)
{
    Super::Initialize(Collection);

    // 연결 관례 2줄 — 문지기를 먼저 초기화시키고 핸들을 빌린다
    UDatabaseSubsystem* Gate = Collection.InitializeDependency<UDatabaseSubsystem>();
    Db = Gate ? Gate->GetDb() : nullptr;
}

FString UArchiveSubsystem::MakeArchiveId(int32 GuildId) const
{
    // 규칙: YYYYMMDDhhmmssSSS-{GUILDID} — ms 정밀로 같은 순간의 충돌을 회피
    const FDateTime Now = FDateTime::UtcNow();   // 시간대 혼선을 피해 항상 UTC
    const FString Stamp = FString::Printf(TEXT("%04d%02d%02d%02d%02d%02d%03d"),
        Now.GetYear(), Now.GetMonth(), Now.GetDay(),
        Now.GetHour(), Now.GetMinute(), Now.GetSecond(), Now.GetMillisecond());

    // 길드 번호는 최소 6자리 zero-pad — 999999를 넘으면 %d가 자릿수를 자연 확장
    return FString::Printf(TEXT("%s-%06d"), *Stamp, GuildId);
}

FString UArchiveSubsystem::CreateArchive(int32 GuildId, const FString& Category,
    ERecordSource Source, int32 DurationSeconds, const FString& Title)
{
    if (!Db || !Db->IsValid()) return FString();   // DB 실패 시 빈 Id — 호출부는 IsEmpty로 판별

    const FString Id = MakeArchiveId(GuildId);
    const FString CreatedAt = FDateTime::UtcNow().ToIso8601();   // TEXT 컬럼엔 ISO8601 문자열로

    FSQLitePreparedStatement Stmt;
    Stmt.Create(*Db, TEXT(
        "INSERT INTO archives (Id, GuildId, Category, Source, CreatedAt, DurationSeconds, Title) "
        "VALUES (?1, ?2, ?3, ?4, ?5, ?6, ?7);"));   // DeletedAt은 넣지 않음 → NULL = 살아있음
    Stmt.SetBindingValueByIndex(1, Id);
    Stmt.SetBindingValueByIndex(2, GuildId);
    Stmt.SetBindingValueByIndex(3, Category);
    Stmt.SetBindingValueByIndex(4, (int32)Source);   // enum → INTEGER 컬럼
    Stmt.SetBindingValueByIndex(5, CreatedAt);
    Stmt.SetBindingValueByIndex(6, DurationSeconds);
    Stmt.SetBindingValueByIndex(7, Title);
    Stmt.Execute();
    Stmt.Destroy();   // 준비된 문은 명시적으로 정리(핸들 누수 방지)

    return Id;
}

TArray<FArchiveRecord> UArchiveSubsystem::GetArchives(int32 GuildId)
{
    TArray<FArchiveRecord> Out;
    if (!Db || !Db->IsValid()) return Out;

    FSQLitePreparedStatement Stmt;
    // DeletedAt IS NULL = 살아있는 기록만(휴지통 제외).
    // (?1 = -1 OR GuildId = ?1) = 쿼리 하나로 "전체 보기"와 "길드 필터"를 겸용
    Stmt.Create(*Db, TEXT(
        "SELECT Id, GuildId, Category, Source, CreatedAt, DurationSeconds, Title "
        "FROM archives WHERE DeletedAt IS NULL AND (?1 = -1 OR GuildId = ?1) "
        "ORDER BY CreatedAt DESC;"));   // 정렬은 날짜 컬럼으로 — 문자열 Id 정렬 금지
    Stmt.SetBindingValueByIndex(1, GuildId);

    while (Stmt.Step() == ESQLitePreparedStatementStepResult::Row)
    {
        FArchiveRecord R;
        int32 SourceInt = 0; FString CreatedAtStr;
        Stmt.GetColumnValueByIndex(0, R.Id);              // 컬럼 인덱스 = SELECT 나열 순서(0부터)
        Stmt.GetColumnValueByIndex(1, R.GuildId);
        Stmt.GetColumnValueByIndex(2, R.Category);
        Stmt.GetColumnValueByIndex(3, SourceInt);
        Stmt.GetColumnValueByIndex(4, CreatedAtStr);
        Stmt.GetColumnValueByIndex(5, R.DurationSeconds);
        Stmt.GetColumnValueByIndex(6, R.Title);

        R.Source = (ERecordSource)SourceInt;                    // INTEGER → enum 복원
        FDateTime::ParseIso8601(*CreatedAtStr, R.CreatedAt);    // TEXT → FDateTime 복원
        Out.Add(R);
    }
    Stmt.Destroy();
    return Out;
}

bool UArchiveSubsystem::SoftDeleteArchive(const FString& Id)
{
    if (!Db || !Db->IsValid()) return false;

    FSQLitePreparedStatement Stmt;
    // 행을 지우지 않는다 — DeletedAt에 시각을 찍으면 "휴지통" 상태(복원 가능).
    // AND DeletedAt IS NULL: 이미 휴지통인 기록의 삭제 시각을 덮어쓰지 않는 가드
    Stmt.Create(*Db, TEXT(
        "UPDATE archives SET DeletedAt = ?2 WHERE Id = ?1 AND DeletedAt IS NULL;"));
    Stmt.SetBindingValueByIndex(1, Id);
    Stmt.SetBindingValueByIndex(2, FDateTime::UtcNow().ToIso8601());
    const bool bOk = Stmt.Execute();
    Stmt.Destroy();
    return bOk;
}

bool UArchiveSubsystem::RestoreArchive(const FString& Id)
{
    if (!Db || !Db->IsValid()) return false;

    FSQLitePreparedStatement Stmt;
    Stmt.Create(*Db, TEXT(
        "UPDATE archives SET DeletedAt = NULL WHERE Id = ?1;"));   // NULL로 되돌리면 부활
    Stmt.SetBindingValueByIndex(1, Id);
    const bool bOk = Stmt.Execute();
    Stmt.Destroy();
    return bOk;
}

⑤ EconomySubsystem — 재화·계정 시간(기능 계층)

    int64 Gold = 0;

    Super::Initialize(Collection);

    UDatabaseSubsystem* Gate = Collection.InitializeDependency<UDatabaseSubsystem>();
    Db = Gate ? Gate->GetDb() : nullptr;

void UEconomySubsystem::AddGold(int64 Delta)
{
    // 음수 방지: 차감이 잔고를 넘으면 0에서 멈춘다(빚 없음 — 정직한 경제)
    Gold = FMath::Max<int64>(0, Gold + Delta);
    SavePlayer();                     // 변화 즉시 저장 — 유실 창 제거
}

void UEconomySubsystem::AddFocusSeconds(int32 Seconds)
{
    if (Seconds <= 0) return;         // 시간은 앞으로만 흐른다 — 음수 적립은 버그 신호
    AccountFocusSeconds += Seconds;   // 레벨은 이 값에서 파생될 뿐, 따로 저장하지 않는다
    SavePlayer();
}

void UEconomySubsystem::SavePlayer()
{
    if (!Db || !Db->IsValid()) return;

    // 현재 값 3종을 봉인 — 평문과 블롭이 "같은 순간의 값"이 되도록 저장 직전에 만든다
    const TArray<uint8> Sealed = SealAccount({ Gold, Shards, AccountFocusSeconds });

    FSQLitePreparedStatement Stmt;
    Stmt.Create(*Db, TEXT(
        "UPDATE player SET Gold = ?1, Shards = ?2, AccountFocusSeconds = ?3, Secured = ?4 WHERE Id = 1;"));
    Stmt.SetBindingValueByIndex(1, Gold);
    Stmt.SetBindingValueByIndex(2, Shards);
    Stmt.SetBindingValueByIndex(3, AccountFocusSeconds);
    Stmt.SetBindingValueByIndex(4, Sealed);          // TArray<uint8> → BLOB 바인딩
    Stmt.Execute();
    Stmt.Destroy();
}

void UEconomySubsystem::LoadPlayer()
{
    int64 PlainGold = 0; int32 PlainShards = 0; int64 PlainFocus = 0;
    TArray<uint8> SecuredBlob;

    FSQLitePreparedStatement Stmt;
    Stmt.Create(*Db, TEXT(
        "SELECT Gold, Shards, AccountFocusSeconds, Secured FROM player WHERE Id = 1;"));
    if (Stmt.Step() == ESQLitePreparedStatementStepResult::Row)
    {
        Stmt.GetColumnValueByIndex(0, PlainGold);
        Stmt.GetColumnValueByIndex(1, PlainShards);
        Stmt.GetColumnValueByIndex(2, PlainFocus);
        Stmt.GetColumnValueByIndex(3, SecuredBlob);   // BLOB → TArray<uint8>
    }
    Stmt.Destroy();

    // [갈래 0] 블롭이 아직 없음(첫 실행) → 평문 채택 후 즉시 봉인 생성
    if (SecuredBlob.Num() == 0)
    {
        Gold = PlainGold; Shards = PlainShards; AccountFocusSeconds = PlainFocus;
        SavePlayer();                                 // 지금 값으로 첫 봉인을 만들어 둔다
        return;
    }

    FSecuredAccount Real;
    if (!OpenAccount(SecuredBlob, Real))
    {
        // [갈래 1] 블롭 자체가 변조/손상 → 진짜 값을 알 수 없음 → 초기화 + 경고
        UE_LOG(LogTemp, Warning, TEXT("[Economy] Secured 손상 — 상태를 초기화합니다"));
        Gold = 0; Shards = 0; AccountFocusSeconds = 0;
        SavePlayer();
        return;
    }

    if (Real.Gold != PlainGold || Real.Shards != PlainShards || Real.AccountFocusSeconds != PlainFocus)
    {
        // [갈래 2] 평문만 편집됨(DB Browser 치트) → 블롭의 진짜 값 채택 = 자동 복원
        UE_LOG(LogTemp, Warning, TEXT("[Economy] 비정상 데이터 감지 — 마지막 정상 값으로 복구"));
        Gold = Real.Gold; Shards = Real.Shards; AccountFocusSeconds = Real.AccountFocusSeconds;
        SavePlayer();                                 // 평문을 즉시 재기록해 정합성 회복
        return;
    }

    // [갈래 3] 일치 → 정상. 평문 그대로 채택
    Gold = PlainGold; Shards = PlainShards; AccountFocusSeconds = PlainFocus;
}
⚠️
빼먹기 쉬운 함정 — 앞으로 player에 보호 값을 새로 추가하면 봉인 페이로드(FSecuredAccount·Seal/Open)에도 같이 넣어야 합니다. 평문만 갱신하면 다음 로드에서 자동 복원이 "정상 저장"을 치트로 오인해 옛 값으로 되돌려버립니다. 모든 쓰기를 SavePlayer() 하나로 통일하세요.

class FSQLiteDatabase;
class UDatabaseSubsystem;   // 트랜잭션 창구(문지기) — 전방 선언으로 충분

    UDatabaseSubsystem* Gate = nullptr; // 트랜잭션 창구 — 서브시스템끼리는 수명이 같아 raw 포인터로 충분

#include "DatabaseSubsystem.h"
#include "SQLiteDatabase.h"
#include "SQLitePreparedStatement.h"

void UGuildSubsystem::Initialize(FSubsystemCollectionBase& Collection)
{
    Super::Initialize(Collection);
    // 문지기 먼저 — 다른 기능 시스템과 완전히 같은 관례. 트랜잭션용으로 Gate도 보관
    Gate = Collection.InitializeDependency<UDatabaseSubsystem>();
    Db = Gate ? Gate->GetDb() : nullptr;
    if (Db && Db->IsValid()) LoadGuilds();

int32 UGuildSubsystem::CreateGuild(const FString& Name, const FString& Category)
{
    if (!Db || !Db->IsValid() || Name.IsEmpty()) return -1;

    FSQLitePreparedStatement Stmt;
    // GuildId는 넣지 않는다 — AUTOINCREMENT가 발급(넣으면 엔진 보장을 스스로 깨는 셈)
    Stmt.Create(*Db, TEXT(
        "INSERT INTO guilds (GuildName, Category, CreatedAt) VALUES (?1, ?2, ?3);"));
    Stmt.SetBindingValueByIndex(1, Name);
    Stmt.SetBindingValueByIndex(2, Category);
    Stmt.SetBindingValueByIndex(3, FDateTime::UtcNow().ToIso8601()); // UTC 고정
    const bool bOk = Stmt.Execute();
    Stmt.Destroy();
    if (!bOk) return -1;

    // 방금 INSERT가 받은 번호 = 이 연결의 마지막 rowid. 이걸 캐시·UI가 쓴다
    const int32 NewId = (int32)Db->GetLastInsertRowId();
    LoadGuilds();                     // 캐시 재적재(단순·확실 — 길드 수가 적어 비용 무시 가능)
    return NewId;
}

void UGuildSubsystem::LoadGuilds()
{
    Guilds.Reset();
    TArray<int32> FixIds;   // 검증 실패로 재기록이 필요한 길드 — SELECT 도중 같은 테이블 UPDATE를 피하려고 모아둔다

    FSQLitePreparedStatement Stmt;
    // 살아있는 길드만(DeletedAt IS NULL), UI 정렬 순서 → 생성 순 보조 정렬
    Stmt.Create(*Db, TEXT(
        "SELECT GuildId, GuildName, Category, FocusSeconds, CreatedAt, SortOrder, Icon, Secured "
        "FROM guilds WHERE DeletedAt IS NULL ORDER BY SortOrder, GuildId;"));
    while (Stmt.Step() == ESQLitePreparedStatementStepResult::Row)
    {
        FGuildData G; FString IsoTime; TArray<uint8> Blob;
        Stmt.GetColumnValueByIndex(0, G.GuildId);
        Stmt.GetColumnValueByIndex(1, G.GuildName);
        Stmt.GetColumnValueByIndex(2, G.Category);
        Stmt.GetColumnValueByIndex(3, G.FocusSeconds);
        Stmt.GetColumnValueByIndex(4, IsoTime);
        FDateTime::ParseIso8601(*IsoTime, G.CreatedAt);   // TEXT → FDateTime
        Stmt.GetColumnValueByIndex(5, G.SortOrder);
        Stmt.GetColumnValueByIndex(6, G.Icon);
        Stmt.GetColumnValueByIndex(7, Blob);              // Secured BLOB

        // ── Secured 3갈래 검증(Economy의 player와 같은 원리 + GuildId 스왑 검사) ──
        int64 Real = 0;
        if (Blob.Num() == 0)
        {
            // [갈래 0] 첫 적립 전 — 통과(첫 AddFocusSeconds 때 봉인이 생성된다)
        }
        else if (!OpenGuildTime(Blob, G.GuildId, Real))
        {
            // [갈래 1] 손상/스왑(다른 길드 블롭) → 진짜 값을 알 수 없음 → 시간 초기화
            UE_LOG(LogTemp, Warning, TEXT("[Guild %d] Secured 손상/불일치 — 시간 초기화"), G.GuildId);
            G.FocusSeconds = 0;
            FixIds.Add(G.GuildId);   // 루프가 끝난 뒤 재봉인 기록
        }
        else if (Real != (int64)G.FocusSeconds)
        {
            // [갈래 2] 평문만 편집됨(치트) → 블롭의 진짜 값으로 자동 복원
            UE_LOG(LogTemp, Warning, TEXT("[Guild %d] 비정상 시간 감지 — 복구"), G.GuildId);
            G.FocusSeconds = (int32)Real;
            FixIds.Add(G.GuildId);
        }
        // [갈래 3] 일치 → 정상, 그대로 사용

        Guilds.Add(G);
    }
    Stmt.Destroy();

    // SELECT가 완전히 끝난 뒤에 재기록 — 조회 중 같은 테이블을 고치는 사고 방지
    for (const int32 Id : FixIds)
        if (const FGuildData* G = Guilds.FindByPredicate([=](const FGuildData& X){ return X.GuildId == Id; }))
            SaveGuildTime(Id, G->FocusSeconds);
}

void UGuildSubsystem::SaveGuildTime(int32 GuildId, int32 FocusSeconds)
{
    if (!Db || !Db->IsValid()) return;

    // 평문+봉인을 같은 UPDATE로 — 어긋난 채 남는 순간이 없게(Economy의 SavePlayer와 동일 원칙)
    const TArray<uint8> Sealed = SealGuildTime(GuildId, FocusSeconds);
    FSQLitePreparedStatement Stmt;
    Stmt.Create(*Db, TEXT("UPDATE guilds SET FocusSeconds = ?1, Secured = ?2 WHERE GuildId = ?3;"));
    Stmt.SetBindingValueByIndex(1, FocusSeconds);
    Stmt.SetBindingValueByIndex(2, Sealed);
    Stmt.SetBindingValueByIndex(3, GuildId);
    Stmt.Execute();
    Stmt.Destroy();
}

void UGuildSubsystem::AddFocusSeconds(int32 GuildId, int32 Seconds)
{
    if (Seconds <= 0) return;         // 시간은 앞으로만
    FGuildData* G = Guilds.FindByPredicate([=](const FGuildData& X){ return X.GuildId == GuildId; });
    if (!G) return;                   // 휴지통 길드엔 적립 불가(캐시 = 살아있는 것만)
    G->FocusSeconds += Seconds;
    SaveGuildTime(GuildId, G->FocusSeconds);
}

int32 UGuildSubsystem::GetGuildLevel(int32 GuildId) const
{
    const FGuildData* G = Guilds.FindByPredicate([=](const FGuildData& X){ return X.GuildId == GuildId; });
    return G ? G->GetLevel() : 0;     // 레벨 = FocusSeconds ÷ 3600 (파생값)
}

void UGuildSubsystem::SoftDeleteGuild(int32 GuildId)
{
    if (!Db || !Db->IsValid()) return;
    FSQLitePreparedStatement Stmt;
    // 행 삭제가 아니라 시각 기록. 이 길드의 기록 행은 건드리지 않는다 —
    // 조회 JOIN의 "g.DeletedAt IS NULL" 조건이 그 기록들을 자동으로 함께 숨긴다(복원하면 함께 복귀)
    Stmt.Create(*Db, TEXT("UPDATE guilds SET DeletedAt = ?1 WHERE GuildId = ?2;"));
    Stmt.SetBindingValueByIndex(1, FDateTime::UtcNow().ToIso8601());
    Stmt.SetBindingValueByIndex(2, GuildId);
    Stmt.Execute();
    Stmt.Destroy();
    LoadGuilds();   // 캐시에서 사라짐(살아있는 것만 담으므로)
}

void UGuildSubsystem::RestoreGuild(int32 GuildId)
{
    if (!Db || !Db->IsValid()) return;
    FSQLitePreparedStatement Stmt;
    Stmt.Create(*Db, TEXT("UPDATE guilds SET DeletedAt = NULL WHERE GuildId = ?1;")); // 되살림 = NULL
    Stmt.SetBindingValueByIndex(1, GuildId);
    Stmt.Execute();
    Stmt.Destroy();
    LoadGuilds();
}

void UGuildSubsystem::AddArchiveCategory(const FString& Name, FLinearColor Color, const FString& Icon)
{
    if (!Db || !Db->IsValid() || Name.IsEmpty()) return;
    FSQLitePreparedStatement Stmt;
    // OR IGNORE: 같은 이름이 이미 있으면 조용히 무시(Name이 PK라 중복은 어차피 거부됨)
    Stmt.Create(*Db, TEXT(
        "INSERT OR IGNORE INTO archive_categories (Name, Color, Icon, CreatedAt) VALUES (?1, ?2, ?3, ?4);"));
    Stmt.SetBindingValueByIndex(1, Name);
    // SQLite엔 색 타입이 없다 — FLinearColor → "RRGGBBAA" 헥스 문자열로 변환해 TEXT 컬럼에
    Stmt.SetBindingValueByIndex(2, Color.ToFColor(true).ToHex());
    Stmt.SetBindingValueByIndex(3, Icon);
    Stmt.SetBindingValueByIndex(4, FDateTime::UtcNow().ToIso8601());
    Stmt.Execute();
    Stmt.Destroy();
}

void UGuildSubsystem::RenameArchiveCategory(const FString& OldName, const FString& NewName)
{
    if (!Db || !Db->IsValid() || !Gate || OldName == NewName || NewName.IsEmpty()) return;

    // 풀(이름 정의)과 기록(태그)이 어긋나면 필터가 깨진다 — 두 UPDATE를 문지기 창구로 묶는다.
    // 중간에 크래시해도 COMMIT 전이면 둘 다 원상 복귀: "풀만 새 이름, 기록은 옛 이름" 상태가 불가능
    if (!Gate->BeginTransaction()) return;

    bool bOk = true;
    {
        FSQLitePreparedStatement S1;
        S1.Create(*Db, TEXT("UPDATE archive_categories SET Name = ?1 WHERE Name = ?2;"));
        S1.SetBindingValueByIndex(1, NewName); S1.SetBindingValueByIndex(2, OldName);
        bOk &= S1.Execute(); S1.Destroy();

        FSQLitePreparedStatement S2;
        S2.Create(*Db, TEXT("UPDATE archives SET Category = ?1 WHERE Category = ?2;"));
        S2.SetBindingValueByIndex(1, NewName); S2.SetBindingValueByIndex(2, OldName);
        bOk &= S2.Execute(); S2.Destroy();
    }
    if (bOk) Gate->CommitTransaction();     // 둘 다 성공 → 한 번에 확정
    else     Gate->RollbackTransaction();   // 하나라도 실패 → 전부 원상 복귀
}

TArray<FArchiveCategory> UGuildSubsystem::GetArchiveCategories() const
{
    TArray<FArchiveCategory> Out;
    if (!Db || !Db->IsValid()) return Out;
    FSQLitePreparedStatement Stmt;
    // 풀 테이블이 정본 — 기록이 없는 "빈 카테고리"도 목록에 나온다(풀 채택의 이유)
    Stmt.Create(*Db, TEXT("SELECT Name, Color, Icon FROM archive_categories ORDER BY SortOrder, Name;"));
    while (Stmt.Step() == ESQLitePreparedStatementStepResult::Row)
    {
        FArchiveCategory C; FString Hex;
        Stmt.GetColumnValueByIndex(0, C.Name);
        Stmt.GetColumnValueByIndex(1, Hex);
        Stmt.GetColumnValueByIndex(2, C.Icon);
        // "RRGGBBAA" 헥스 TEXT → FLinearColor 복원(저장 때의 역변환)
        C.Color = FLinearColor::FromSRGBColor(FColor::FromHex(Hex));
        Out.Add(C);
    }
    Stmt.Destroy();
    return Out;
}
